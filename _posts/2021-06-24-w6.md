```python
list1 = [1,2]
int1 = 0
float1 = 3.14
set1 = set()
tuple1 = (2,3)
    
print(id(list1))
print(id(int1))
print(id(float1))
print(id(set1))
print(id(tuple1))

# 1926040931200
# 1926035368208
# 1926043862544
# 1926041215712
```
